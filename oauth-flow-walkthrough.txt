Function: getToken(mcpUrl)

  // First checks in-memory Map
  if (!this.tokens.has(mcpUrl)) this.loadFromDb(mcpUrl);
  //   ↑ If not in memory, tries to load from SQLite

  // loadFromDb() calls getTokenForUrl() in db.ts which runs:
  //   SELECT access_token, refresh_token, token_expires_at, token_endpoint, client_id
  //   FROM connections WHERE url = ? AND access_token IS NOT NULL

  // If token found but near expiry (within 5 minutes):
  const almostExpired = tok.expiresAt && (Date.now() > tok.expiresAt - 5 * 60_000);
  if (almostExpired) {
    let refreshed = await this.refreshAccessToken(tok);
    if (!refreshed) refreshed = await this.refreshAccessToken(tok); // retry once
    // refreshAccessToken() does:
    //   POST {tok.tokenEndpoint}
    //   grant_type=refresh_token&refresh_token={tok.refreshToken}&client_id={tok.clientId}
    // Updates tok.accessToken and tok.expiresAt in place if successful
  }

  return tok.accessToken;  // or null if no token / refresh failed

  // Pattern-match common auth error formats from different MCP servers:
  //   Granola returns: 401 Unauthorized
  //   Notion returns:  {"error":"invalid_token","error_description":"Missing or invalid access token"}
  //   Others may return: 403, "forbidden", etc.

  const isAuthError = transport === "http" && (
    msg.includes("401") || msg.includes("unauthorized") || msg.includes("403") ||
    msg.includes("invalid_token") || msg.includes("token") || msg.includes("auth") ||
    msg.includes("forbidden") || !oauthManager.hasTokenSync(url || "")
  );

  // If auth error detected, tell the frontend:
  if (isAuthError) {
    return NextResponse.json({ error: error.message, needsAuth: true }, { status: 401 });
    //                                                 ^^^^^^^^^^^^
    //                                     This flag is what triggers the OAuth flow
  }

    const { mcpUrl } = await req.json();

    // Build callback URL from the current request origin

    const origin = new URL(req.url).origin;

    const { authUrl, state } = await oauthManager.startFlow(mcpUrl, callbackUrl);

File: src/lib/oauth.ts
Function: startFlow(mcpUrl, callbackUrl)

  // ── A. DISCOVERY ──
  // Try to find the OAuth server's configuration.
  // Attempts three well-known URLs in order:

  async discoverMetadata(mcpUrl: string) {
    const origin = new URL(mcpUrl).origin;
    // e.g. for Granola: origin = "https://mcp.granola.ai"

    // Try 1: MCP-spec protected resource metadata
    //   GET https://mcp.granola.ai/.well-known/oauth-protected-resource
    //   Response includes: { authorization_servers: ["https://mcp-auth.granola.ai"] }
    //   Then fetch: GET https://mcp-auth.granola.ai/.well-known/oauth-authorization-server
    //   Response includes: { authorization_endpoint, token_endpoint, registration_endpoint }

    // Try 2: Direct well-known on the origin
    //   GET https://mcp.granola.ai/.well-known/oauth-authorization-server

    // Try 3: OpenID configuration
    //   GET https://mcp.granola.ai/.well-known/openid-configuration

    // Returns: { authorization_endpoint, token_endpoint, registration_endpoint }
  }

  // ── B. DYNAMIC CLIENT REGISTRATION ──
  // Our app doesn't have pre-registered credentials with Granola/Notion.
  // So we register ourselves on the fly.

  if (metadata.registration_endpoint) {
    const res = await fetch(metadata.registration_endpoint, {
      method: "POST",
      headers: { "Content-Type": "application/json" },
      body: JSON.stringify({
        client_name: "MCP Tester",
        redirect_uris: [callbackUrl],   // e.g. http://localhost:3000/api/mcp/oauth/callback
        grant_types: ["authorization_code"],
        response_types: ["code"],
        token_endpoint_auth_method: "none",  // Public client, no secret needed
      }),
    });
    // Response: { client_id: "client_01KH773..." }
    clientId = data.client_id;
  }

  // ── C. PKCE (Proof Key for Code Exchange) ──
  // Prevents anyone who intercepts the auth URL from completing the flow.

  const verifier = crypto.randomBytes(32).toString("base64url");
  // e.g. "dBjftJeZ4CVP-mB92K27uhbUJU1p1r_wW1gFWFOEjXk"

  const challenge = crypto.createHash("sha256").update(verifier).digest("base64url");
  // e.g. "E9Melhoa2OwvFrEMTJguCHaoeK1t8URWbuGJSstw-cM"

  const state = crypto.randomBytes(16).toString("hex");
  // e.g. "a3f5b2c1d4e6f7890123456789abcdef"

  // ── D. STORE STATE SERVER-SIDE ──
  // We need the verifier later (in the callback) to prove we started this flow.
  this.pending.set(state, {
    codeVerifier: verifier,   // Secret — never sent to the browser
    state,
    clientId,
    redirectUri: callbackUrl,
    tokenEndpoint: metadata.token_endpoint,
    mcpUrl,
  });

  // ── E. BUILD AUTH URL ──
  // This is what the browser popup will open.
  const params = new URLSearchParams({
    response_type: "code",
    client_id: clientId,
    redirect_uri: callbackUrl,
    state: state,
    code_challenge: challenge,        // Hash of verifier (public)
    code_challenge_method: "S256",
  });

  return {
    authUrl: `${metadata.authorization_endpoint}?${params}`,
    // e.g. "https://mcp-auth.granola.ai/oauth2/authorize?response_type=code&client_id=...&..."
  };

File: src/lib/oauth.ts
Function: exchangeCode(state, code)

  // Look up the pending auth by state
  const pending = this.pending.get(state);
  // This gives us back: codeVerifier, clientId, redirectUri, tokenEndpoint, mcpUrl

  // ── TOKEN EXCHANGE ──
  // Send the authorization code + our PKCE verifier to get tokens.
  // The server checks: SHA256(verifier) === the challenge we sent in Step 5.
  // This proves we're the same app that started the flow.

  const res = await fetch(pending.tokenEndpoint, {
    method: "POST",
    headers: { "Content-Type": "application/x-www-form-urlencoded" },
    body: new URLSearchParams({
      grant_type: "authorization_code",
      code: code,                        // One-time code from the redirect
      redirect_uri: pending.redirectUri,  // Must match what we registered
      client_id: pending.clientId,
      code_verifier: pending.codeVerifier, // PKCE proof
    }).toString(),
  });

  // Response from Granola:
  // {
  //   "access_token": "eyJhbGciOiJSUzI1NiI...",   ← This is what we need
  //   "refresh_token": "oUKBdRg1UlnM7nOXteu...",   ← For getting new access tokens later
  //   "expires_in": 21600,                           ← Seconds until access_token expires (6 hours)
  //   "token_type": "Bearer"
  // }

  // ── STORE TOKEN ──
  // Stored in memory (Map keyed by MCP URL) with all metadata needed for refresh
  this.tokens.set(pending.mcpUrl, {
    accessToken: data.access_token,
    refreshToken: data.refresh_token,
    expiresAt: Date.now() + data.expires_in * 1000,
    tokenEndpoint: pending.tokenEndpoint,   // Needed later for refresh
    clientId: pending.clientId,             // Needed later for refresh
  });

  // Clean up — this state can't be reused
  this.pending.delete(state);

========================================
SUMMARY: Where each piece lives
========================================

  Frontend (page.tsx):
    - Opens popup for user login
    - Listens for oauth_success message
    - Retries connection after auth

  /api/mcp/oauth/start (start/route.ts):
    - Discovers OAuth endpoints
    - Registers as OAuth client
    - Generates PKCE
    - Returns auth URL for popup

  /api/mcp/oauth/callback (callback/route.ts):
    - Receives authorization code from provider redirect
    - Exchanges code for access_token + refresh_token
    - Closes popup

  OAuth manager (oauth.ts):
    - In-memory token store (Map<mcpUrl, token>)
    - PKCE generation
    - OAuth discovery (3 well-known URLs)
    - Dynamic client registration
    - Token exchange
    - Token refresh (automatic, 5-min buffer, 1 retry)
    - Load from / persist to SQLite

  DB (db.ts):
    - SQLite table with token columns
    - Tokens survive server restarts

  MCP manager (mcp-manager.ts):
    - Attaches Bearer token to HTTP headers
    - Uses token when calling MCP tools

  Connect route (connect/route.ts):
    - Checks for token before connecting
    - Detects auth errors (401, invalid_token, etc.)
    - Returns { needsAuth: true } to trigger OAuth

  Chat route (chat/route.ts):
    - Never touches OAuth directly
    - Just calls mcpManager.callTool() which uses the already-authed connection
